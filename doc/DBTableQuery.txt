CREATE TABLE `user` (
  `USERID` int(11) NOT NULL AUTO_INCREMENT,
  `LOGINID` varchar(45) COLLATE utf8mb4_general_ci NOT NULL,
  `PASSWORD` varchar(100) COLLATE utf8mb4_general_ci NOT NULL,
  `CREATEDATE` datetime DEFAULT NULL,
  `ROLE` int(11) DEFAULT NULL,
  PRIMARY KEY (`USERID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci


CREATE TABLE `team` (
  `TEAMID` int(11) NOT NULL AUTO_INCREMENT,
  `USERID` int(11) NOT NULL,
  `TEAMNAME` varchar(100) COLLATE utf8mb4_general_ci NOT NULL,
  `CREATEDATE` datetime DEFAULT NULL,
  PRIMARY KEY (`TEAMID`),
  KEY `USERID` (`USERID`),
  CONSTRAINT `USERID` FOREIGN KEY (`USERID`) REFERENCES `user` (`USERID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci

CREATE TABLE `schedule` (
  `SCHEDULEID` int(11) NOT NULL AUTO_INCREMENT,
  `USERID` int(11) DEFAULT NULL,
  `SCHEDULENAME` varchar(50) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `STARTDATE` datetime DEFAULT NULL,
  `ENDDATE` datetime DEFAULT NULL,
  `PREWORKDATE` decimal(5,2) DEFAULT NULL,
  `COMPLETION` int(11) DEFAULT NULL,
  `WORKDATE` decimal(5,2) DEFAULT NULL,
  PRIMARY KEY (`SCHEDULEID`),
  KEY `SCHEDULEtoUSER` (`USERID`),
  CONSTRAINT `SCHEDULEtoUSER` FOREIGN KEY (`USERID`) REFERENCES `user` (`USERID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci

CREATE TABLE `schedule_time` (
  `TIMEID` int(11) NOT NULL AUTO_INCREMENT,
  `USERID` int(11) DEFAULT NULL,
  `SCHEDULEID` int(11) DEFAULT NULL,
  `STARTDATE` datetime DEFAULT NULL,
  `ENDDATE` datetime DEFAULT NULL,
  `PAUSEDATE` datetime DEFAULT NULL,
  `PASECOMMENT` varchar(1000) COLLATE utf8mb4_general_ci DEFAULT NULL,
  PRIMARY KEY (`TIMEID`),
  KEY `SCHEDULE_TIMEtoUSER` (`USERID`),
  KEY `SCHEDULE_TIMEtoSCHEDULE` (`SCHEDULEID`),
  CONSTRAINT `SCHEDULE_TIMEtoSCHEDULE` FOREIGN KEY (`SCHEDULEID`) REFERENCES `schedule` (`SCHEDULEID`),
  CONSTRAINT `SCHEDULE_TIMEtoUSER` FOREIGN KEY (`USERID`) REFERENCES `user` (`USERID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci
